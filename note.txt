SELECT DISTINCT Entities.EntityID
FROM Entities

    JOIN EntityValues AS EV1
    ON Entities.EntityID = EV1.EntityID

    JOIN EntityValues AS EV2
    ON Entities.EntityID = EV2.EntityID

WHERE Entities.EntityType = 'Product' AND EV1.Attribute = 'name' AND EV1.Value = 'iphone1' AND EV2.Attribute = 'price' AND EV2.Value = '1'
